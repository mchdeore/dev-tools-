local hyper = {"ctrl", "alt", "cmd"}

hs.window.animationDuration = 0

local GRID_COLUMNS = 6
local GRID_ROWS = 6

local function focusedWin()
  return hs.window.focusedWindow()
end

local function clamp(n, minVal, maxVal)
  if n < minVal then return minVal end
  if n > maxVal then return maxVal end
  return n
end

-- Build exact pixel boundaries for a grid.
-- This avoids float drift and guarantees the grid spans the screen cleanly.
local function makeBoundaries(startPos, totalSize, divisions)
  local out = {}
  for i = 0, divisions do
    out[i] = math.floor(startPos + (totalSize * i / divisions) + 0.5)
  end
  return out
end

-- Find the nearest grid boundary index for a pixel coordinate.
local function nearestBoundaryIndex(boundaries, value, maxIndex)
  local bestIndex = 0
  local bestDist = math.huge

  for i = 0, maxIndex do
    local d = math.abs(value - boundaries[i])
    if d < bestDist then
      bestDist = d
      bestIndex = i
    end
  end

  return bestIndex
end

-- Convert a window frame into grid coordinates by snapping all 4 edges.
local function getGridFrame(win)
  local screen = win:screen():frame()
  local f = win:frame()

  local xBounds = makeBoundaries(screen.x, screen.w, GRID_COLUMNS)
  local yBounds = makeBoundaries(screen.y, screen.h, GRID_ROWS)

  local left = nearestBoundaryIndex(xBounds, f.x, GRID_COLUMNS)
  local right = nearestBoundaryIndex(xBounds, f.x + f.w, GRID_COLUMNS)
  local top = nearestBoundaryIndex(yBounds, f.y, GRID_ROWS)
  local bottom = nearestBoundaryIndex(yBounds, f.y + f.h, GRID_ROWS)

  -- Ensure at least 1 cell in each dimension
  if right <= left then right = math.min(GRID_COLUMNS, left + 1) end
  if bottom <= top then bottom = math.min(GRID_ROWS, top + 1) end

  local gx = clamp(left, 0, GRID_COLUMNS - 1)
  local gy = clamp(top, 0, GRID_ROWS - 1)
  local gw = clamp(right - gx, 1, GRID_COLUMNS - gx)
  local gh = clamp(bottom - gy, 1, GRID_ROWS - gy)

  return {
    screen = screen,
    xBounds = xBounds,
    yBounds = yBounds,
    gx = gx,
    gy = gy,
    gw = gw,
    gh = gh
  }
end

local function applyGridFrame(win, g)
  local x1 = g.xBounds[g.gx]
  local x2 = g.xBounds[g.gx + g.gw]
  local y1 = g.yBounds[g.gy]
  local y2 = g.yBounds[g.gy + g.gh]

  win:setFrame({
    x = x1,
    y = y1,
    w = x2 - x1,
    h = y2 - y1
  })
end

-- Move window by grid cells without changing size
local function moveWindow(dx, dy)
  local win = focusedWin()
  if not win then return end

  local g = getGridFrame(win)

  g.gx = clamp(g.gx + dx, 0, GRID_COLUMNS - g.gw)
  g.gy = clamp(g.gy + dy, 0, GRID_ROWS - g.gh)

  applyGridFrame(win, g)
end

-- Resize from the RIGHT edge only
local function resizeWidth(delta)
  local win = focusedWin()
  if not win then return end

  local g = getGridFrame(win)

  g.gw = clamp(g.gw + delta, 1, GRID_COLUMNS - g.gx)

  applyGridFrame(win, g)
end

-- Resize from the BOTTOM edge only
local function resizeHeight(delta)
  local win = focusedWin()
  if not win then return end

  local g = getGridFrame(win)

  g.gh = clamp(g.gh + delta, 1, GRID_ROWS - g.gy)

  applyGridFrame(win, g)
end

-- Center window while preserving size
local function centerWindow()
  local win = focusedWin()
  if not win then return end

  local g = getGridFrame(win)

  g.gx = math.floor((GRID_COLUMNS - g.gw) / 2)
  g.gy = math.floor((GRID_ROWS - g.gh) / 2)

  applyGridFrame(win, g)
end

-- Fill screen exactly
local function maximizeWindow()
  local win = focusedWin()
  if not win then return end

  local g = getGridFrame(win)
  g.gx = 0
  g.gy = 0
  g.gw = GRID_COLUMNS
  g.gh = GRID_ROWS

  applyGridFrame(win, g)
end

local function minimizeWindow()
  local win = focusedWin()
  if win then win:minimize() end
end

-- Optional helpers for quick tiling
local function leftHalf()
  local win = focusedWin()
  if not win then return end
  local g = getGridFrame(win)
  g.gx = 0
  g.gy = 0
  g.gw = 3
  g.gh = GRID_ROWS
  applyGridFrame(win, g)
end

local function rightHalf()
  local win = focusedWin()
  if not win then return end
  local g = getGridFrame(win)
  g.gx = 3
  g.gy = 0
  g.gw = 3
  g.gh = GRID_ROWS
  applyGridFrame(win, g)
end

-- Move
hs.hotkey.bind(hyper, "Left", function() moveWindow(-1, 0) end)
hs.hotkey.bind(hyper, "Right", function() moveWindow(1, 0) end)
hs.hotkey.bind(hyper, "Up", function() moveWindow(0, -1) end)
hs.hotkey.bind(hyper, "Down", function() moveWindow(0, 1) end)

-- Horizontal resize
hs.hotkey.bind(hyper, "/", function() resizeWidth(-1) end)   -- shrink
hs.hotkey.bind(hyper, "'", function() resizeWidth(1) end)    -- grow

-- Vertical resize
hs.hotkey.bind(hyper, ".", function() resizeHeight(-1) end)  -- shrink
hs.hotkey.bind(hyper, ";", function() resizeHeight(1) end)   -- grow

-- Core controls
hs.hotkey.bind(hyper, "C", centerWindow)
hs.hotkey.bind(hyper, "M", maximizeWindow)
hs.hotkey.bind(hyper, "\\", minimizeWindow)

-- Optional quick halves
hs.hotkey.bind(hyper, "[", leftHalf)
hs.hotkey.bind(hyper, "]", rightHalf)

-- Move to next monitor and preserve approximate grid placement
hs.hotkey.bind(hyper, "N", function()
  local win = focusedWin()
  if not win then return end

  local oldGrid = getGridFrame(win)
  local nextScreen = win:screen():next()
  win:moveToScreen(nextScreen, false, true)

  local newScreen = nextScreen:frame()
  oldGrid.screen = newScreen
  oldGrid.xBounds = makeBoundaries(newScreen.x, newScreen.w, GRID_COLUMNS)
  oldGrid.yBounds = makeBoundaries(newScreen.y, newScreen.h, GRID_ROWS)

  applyGridFrame(win, oldGrid)
end)

-- Launch Terminal
hs.hotkey.bind(hyper, "T", function()
  hs.application.launchOrFocus("Terminal")
end)
